# Scope

- [] Database:
    - [] entrepreneur
    - [] message
    - [] status_message
    - [] segmentation
- [] entrepreneur:
    - id: id
    - name: str
    - phone_number: str
    - segmentation: str
    - historic: historic
- [] message:
    - id: id
    - created_at: date
    - name_template: str
    - name->segmantation: str
    - status_message: status_message
- [] segmantation:
    - id: id
    - name: str
    - updated_at: date
- [] status_message(enum):
    - created_at: date
    - updated_at: date
    - not sended: str
    - sended: str
    - delivered: str
    - readed: str
- [] historic:
    - id: id
    - id_message: int
    - id_entrepreneur: int
